ROI of Travel: For high-mileage categories like "Customer Visit," can we correlate travel frequency with client retention or revenue growth?¶

In [36]:
# Filter and aggregate customer visits  
client_visits = df[df['PURPOSE'] == 'Customer Visit']  
visit_analysis = client_visits.groupby('STOP').agg(  
    TOTAL_MILES=('MILES', 'sum'),  
    TRIP_COUNT=('MILES', 'count')  
).reset_index().sort_values('TOTAL_MILES', ascending=False).head(10)  

# Plotting  
fig, (ax1, ax2) = plt.subplots(1, 2, figsize=(18, 6))  

# Total Miles per Client Location  
ax1.barh(visit_analysis['STOP'], visit_analysis['TOTAL_MILES'], color='skyblue')  
ax1.set_title('Total Miles per Client Location')  
ax1.set_xlabel('Miles')  
ax1.invert_yaxis()  

# Trip Count per Client Location  
ax2.barh(visit_analysis['STOP'], visit_analysis['TRIP_COUNT'], color='salmon')  
ax2.set_title('Trip Frequency per Client Location')  
ax2.set_xlabel('Trips')  
ax2.invert_yaxis()  

plt.tight_layout()  
plt.savefig("Trip Frequency per Client Location.png")
plt.show()
